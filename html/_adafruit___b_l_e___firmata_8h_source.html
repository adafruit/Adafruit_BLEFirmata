48 // message command bytes (128-255/0x80-0xFF)
49 #define DIGITAL_MESSAGE 0x90 // send data for a digital pin

52 #define REPORT_DIGITAL 0xD0 // enable digital input by port pair

59 #define START_SYSEX 0xF0 // start a MIDI Sysex message

62 // extended command set using sysex (0-127/0x00-0x7F)
63 /* 0x00-0x0F reserved for user-defined commands */
64 #define SERVO_CONFIG 0x70 // set max angle, minPulse, maxPulse, freq
65 #define STRING_DATA 0x71 // a string message with 14-bits per char
66 #define SHIFT_DATA 0x75 // a bitstream to/from a shift register
67 #define I2C_REQUEST 0x76 // send an I2C read/write request
68 #define I2C_REPLY 0x77 // a reply to an I2C read request
69 #define I2C_CONFIG \
70  0x78 // config I2C settings such as delay times and power pins
71 #define EXTENDED_ANALOG 0x6F // analog write (PWM, Servo, etc) to any pin
72 #define PIN_STATE_QUERY 0x6D // ask for a pin's current mode and value
73 #define PIN_STATE_RESPONSE 0x6E // reply with pin's current mode and value
74 #define CAPABILITY_QUERY \
75  0x6B // ask for supported modes and resolution of all pins
76 #define CAPABILITY_RESPONSE 0x6C // reply with supported modes and resolution
77 #define ANALOG_MAPPING_QUERY 0x69 // ask for mapping of analog to pin numbers
78 #define ANALOG_MAPPING_RESPONSE 0x6A // reply with mapping info
79 #define REPORT_FIRMWARE 0x79 // report name and version of the firmware
80 #define SAMPLING_INTERVAL 0x7A // set the poll rate of the main loop
81 #define SYSEX_NON_REALTIME 0x7E // MIDI Reserved for non-realtime messages
82 #define SYSEX_REALTIME 0x7F // MIDI Reserved for realtime messages
83 // these are DEPRECATED to make the naming more consistent

85 #define SYSEX_I2C_REQUEST 0x76 // same as I2C_REQUEST
